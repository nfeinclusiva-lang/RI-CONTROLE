import streamlit as st import pandas as pd import plotly.express as px # Configuração da página st.set_page_config(page_title="Meu Dashboard Financeiro", layout="wide") # Link que você me enviou (formato CSV) SHEET_URL = "https://docs.google.com/spreadsheets/d/e/2PACX-1vT3u3PVaORUFqh2jnvRvN9WixGWGwqtaEx1Nt6wviX_PXstOvDD83SxAt0RR4X4C1s1Gl9bpfCqK8fz/pub?gid=1436090432&single=true&output=csv" @st.cache_data def load_data(): # Lê os dados direto do link df = pd.read_csv(SHEET_URL) # Ajuste de nomes de colunas (caso necessário, o pandas limpa os espaços) df.columns = [c.strip() for c in df.columns] # Criando a coluna de Saldo (Orçado - Gasto) # Nota: Certifique-se que na sua planilha as colunas se chamam 'Orcado' e 'Gasto' if 'Orcado' in df.columns and 'Gasto' in df.columns: df['Saldo'] = df['Orcado'] - df['Gasto'] return df try: df = load_data() st.title("📊 Dashboard Financeiro Dinâmico") st.markdown(f"Dados sincronizados via Google Sheets") # --- MÉTRICAS PRINCIPAIS --- total_orcado = df['Orcado'].sum() total_gasto = df['Gasto'].sum() total_saldo = total_orcado - total_gasto c1, c2, c3 = st.columns(3) c1.metric("Valor Orçado", f"R$ {total_orcado:,.2f}") c2.metric("Valor Gasto", f"R$ {total_gasto:,.2f}", delta=f"{((total_gasto/total_orcado)-1)*100:.1f}%", delta_color="inverse") c3.metric("Saldo Disponível", f"R$ {total_saldo:,.2f}") st.divider() # --- GRÁFICOS --- col_left, col_right = st.columns(2) with col_left: st.subheader("Comparativo por Item") fig_comp = px.bar(df, x="Item", y=["Orcado", "Gasto"], barmode="group", color_discrete_map={"Orcado": "#3498db", "Gasto": "#e74c3c"}) st.plotly_chart(fig_comp, use_container_width=True) with col_right: st.subheader("Análise de Saldo por Item") # Mostra quanto sobrou ou faltou em cada categoria fig_saldo = px.bar(df, x="Item", y="Saldo", color="Saldo", color_continuous_scale="RdYlGn") st.plotly_chart(fig_saldo, use_container_width=True) # --- TABELA DETALHADA --- st.subheader("Tabela de Dados Completa") st.dataframe(df, use_container_width=True) except Exception as e: st.error(f"Erro ao carregar dados: {e}") st.info("Dica: Verifique se os nomes das colunas na planilha são 'Item', 'Orcado' e 'Gasto'.")
